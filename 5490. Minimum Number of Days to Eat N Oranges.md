class Solution:
    def dfs(self, n: int, dp) -> int:
        # print(n)
        if n in dp.keys():
            return dp[n]
        dp[n] = min(self.dfs(n//3,dp)+n%3+1,self.dfs(n//2,dp)+n%2+1)
        return dp[n]
        
    def minDays(self, n: int) -> int:
        dp = {1:1,0:0}
        return self.dfs(n,dp)
